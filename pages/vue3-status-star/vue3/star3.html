# 1) setup

  • 新的option, 所有的组合API函数都在此使用, 只在初始化时执行一次
  • 函数如果返回对象, 对象中的属性或方法, 模板中可以直接使用

# 2) ref

  • 作用: 定义一个数据的响应式(基本类型)
  • 语法: const xxx = ref(initValue):
    • 创建一个包含响应式数据的引用(reference)对象
    • js中操作数据: xxx.value
    • 模板中操作数据: 不需要.value
  • 一般用来定义一个基本类型的响应式数据
<template>
   <h2>setup和ref基本使用</h2>
  <h3>{{ count }}</h3>
  <!-- <h3>{{ count.value }}</h3> -->
  <button @click="updateCount">点击更新数据</button>
</template>

<script>
// 引入 ref 
import { ref } from 'vue'

export default {

  name: 'App',
  //  页面打开后 可以看到一个数据 点击按钮后 数据可以发生变化

  // vue2
  // data() {
  //   return {
  //     count: '777',
  //   }
  // },
  // methods: {
  //   updateCount() {
  //     this.count++
  //   },
  // },

  // vue3
  // setup是组合api的入口

  setup () {

   // 变量
    // let count = 0 // 此时数据不是响应式数据
    //  ref是定义一个响应式数据,返回的是一个ref对象 对象中有一个value属性 如果需要对数据进行操作，需要使用该Ref对象调用value属性的方式进行数据操作
    //  html模板中不需要使用.value属性的写法
    //  一般用来定义基本数据的响应式类型
    //  此时的count 是Ref类型
    const count = ref(0)
    console.log(count)
    //  返回的是一个对象
    function updateCount() {
      console.log('-----')
      // count++  //报错原因： count是一个对象 对象不能++
      count.value++
    }
    return {
      //  属性
      count,
      // 方法
      updateCount,
    }
  }
}
</script>

# 3) reactive

  • 作用: 定义多个数据的响应式
  • const proxy = reactive(obj): 接收一个普通对象然后返回该普通对象的响应式代理器对象
  • 响应式转换是“深层的”：会影响对象内部所有嵌套的属性
  • 内部基于 ES6 的 Proxy 实现，通过代理对象操作源对象内部数据都是响应式的
<template>
  <h2>reactive的使用</h2>
  <h3>姓名：{{ user.name }}</h3>
  <h3>年龄：{{ user.age }}</h3>
  <h3>对象：{{ user.wife }}</h3>
  <h3>性别：{{ user.gender }}</h3>
  <button @click="updateUser">点击切换数据</button>
</template>

<script lang="ts">
import { reactive } from 'vue'

export default defineComponent({
  name: 'App',
  //  需求：显示用户的相关信息，点击按钮可以更新用户相关信息数据

  // -----
  // const proxy = reactive(obj): 接收一个普通对象然后返回该普通对象的响应式代理器对象
  // 响应式转换是“深层的”：会影响对象内部所有嵌套的属性
  // 内部基于 ES6 的 Proxy 实现，通过代理对象操作源对象内部数据都是响应式的
  setup() {
    console.log(1111)
    //  把数据变成响应式数据
    //  返回的是一个Proxy的代理对象，被代理者的目标对象就是obj对象  reactive(obj)
    //  user现在是代理对象，obj是被代理对象
    //  user对象的类型是Proxy
    // const obj: any = { // 为了在使用obj.gender='男' 的时候 不出现错误信息才写的
    const obj = {
      name: 'pingan',
      age: 18,
      wife: {
        name: 'yuanyuan',
        age: 17,
        cars: ['奔驰', '宝马', '路虎'],
      },
    }
    const user = reactive<any>(obj)
    console.log(user)
    //  方法
    // function updateUser (){}
    const updateUser = () => {
      //  直接使用目标对象的方式来更新目标对象中成员的的值 是不可能的
      //  只能使用代理对象的方式来更新数据 （响应式数据）
      // obj.name += '==='
      //
      // user.name += '平安'
      // user.age += 1
      // user.wife.name += 'yuan'
      // user.wife.cars[0] = '玛莎'
      //  user---代理对象   obj---目标对象
      //  user对象或者obj对象添加新的属性  哪一种方式会影响页面更新
      //  obj.gender = '男' // 这种方式界面没有更新渲染
      // user.gender = '男' // 这种方式界面可以更新渲染，而且数据最终添加到了obi对象上了
      //  user对象或者obj对象移除一个属性  哪一种方式会影响页面更新
      // delete obj.age // 这个界面没有更新渲染，但是obj中确实没有了age这个属性
      // delete user.age // 这个界面更新渲染了，但是obj中确实没有了age这个属性
      //  如果操作代理对象，目标对象中的数据也会随之变化，同时如果想要在操作数据的时候，界面也要跟着重现渲染，那么也是操作代理对象
      //  通过当前代理对象  找到该对象中的某个属性，更改该属性中的某个数据
      // user.wife.cars[1] = '迈巴赫'
      // 添加一个数据
      user.wife.cars[3] = '皮卡'
    }

    return {
      user,
      updateUser,
    }
  },
})
</script>
